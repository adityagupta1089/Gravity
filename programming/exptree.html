#include <iostream>

using namespace std;

#define M1 1000000007
#define M2 1000000009

typedef long long ll;

int gcd(ll a, ll b) {
    return b == 0 ? a : gcd(b, a % b);
}

ll modPow(ll x, int n, int p){
    if(!n) return 1;
    else if(n&1)return (x*modPow(x,n-1,p))%p;
    else {
        ll val = modPow(x, n/2, p);
        return (val*val)%p;
    }
}

int main() {
    int t;
    cin >> t;
    while(t--){
        ll n;
        cin >> n;
        ll n1 = n%M1;
        ll p1 = (n1*(n1-1+M1))%M1;
        ll q1 = (4*n1-6+M1);
        ll cd = gcd(p1,q1);
        p1/=cd;q1/=cd;
        cout << (p1*modPow(q1,M1-2,M1))%M1 << " ";

        ll n2 = n % M2;
        ll p2 = (n2*(n2-1+M2))%M2;
        ll q2 = (4*n2-6+M2);
        ll cd2 = gcd(p2,q2);
        p2/=cd2;q2/=cd2;
        cout << (p2*modPow(q2,M2-2,M2))%M2 << "\n";
    }
}
